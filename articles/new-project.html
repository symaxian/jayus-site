Local Project

For a local project the small html file below is all you need to get started. It must be placed in the same directory(folder) holding the jayus project folder. 

<html>
    <head>
        <title>New Jayus Project</title>
        <script type="text/javascript" src="jayus/jayus.js"></script>
        <script type="text/javascript">

            // This function will be called when the page(and jayus) is done loading
            function init(){
            
                // Initialize jayus
                jayus.init();

                // Create a new display(a visible canvas)
                var display = new jayus.Display(800, 600);

                // Set the display on the page
                display.setInto(document.getElementById('jayus'));

                // Set a css border around the display
                display.canvas.style.border = '1px solid grey';


                // Put your initialization code here


                // Add an event handler for the frame event
                jayus.addHandler('frame', function(event){


                    // Any code you put here will be executed every frame


                });


                // Start jayus
                jayus.start();

            }

        </script>
    </head>
    <body onload="init()">
        <center id="jayus"> <!-- Our display container --> </center>
    </body>
</html>
